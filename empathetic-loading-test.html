Implementation Guide

1. Include the script
<script src="/static/js/empathetic-loading.js"></script>
2. Create a container element
<div id="loading-container"></div>
3. Initialize and control the loader
// Create the loader
const loader = window.EmpatheticLoading.create({
  targetElementId: 'loading-container',
  context: 'documentGeneration',
  duration: 5000,
  showProgressBar: true,
  showSupportiveMessages: true,
  onComplete: function() {
    console.log('Loading complete!');
  }
});

// Start the loader
loader.start();

// Update manually (optional)
loader.update({
  percentage: 50,
  message: "Custom update message",
  supportiveMessage: "You're doing great!"
});

// Complete when done
loader.complete("Your document has been successfully generated!");

// Reset if needed
loader.reset();
